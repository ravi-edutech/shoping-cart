import React from "react";
import { render, screen } from '@testing-library/react';
import userPreferencesReducer from "./userPreferences";
import Types from "../actions/actionTypes";
 
test('CATEGORY_SELECTED', () => {
    const INITIAL_STATE = {
        'selectedCategory':'',
        'selectedProduct': {},
        'selectedLanguage': 'English',
        'cart':[]
    }
    const categories = userPreferencesReducer(INITIAL_STATE,Types.CATEGORY_SELECTED)
 
    expect(categories['selectedCategory']).toHaveLength(0);
});
 
 
test('LANGUAGE_CHANGED', () => {

        'selectedProduct': {},
        'selectedLanguage': 'English',
        'cart':[]
    }
    const categories = userPreferencesReducer(INITIAL_STATE,Types.LANGUAGE_CHANGED)
 
    expect(categories['selectedLanguage']).toHaveLength(7);
});
